title: Bestellungen
description: |+
  Dieser Bericht erzeugt einige Bestelllisten zur Überwachung der aktuellen
  Bestellverarbeitung.

query:
  tables:
    database: dvb_manage_live
    command: >
      SELECT    table_schema, table_name
      FROM      information_schema.tables
      ORDER BY  table_schema, table_name

  full-BT:
    title: 'reine BT Bestellungen'
    description: listet Bestellungen älter als drei Tage, in denen nur B&T-Titel enthalten sind
    database: dvb_manage_live
    command: "
    select l.name, lo.description, lo.order_date
    from mng_library_order lo
    inner join mng_library l on l.library_id = lo.library_id
    where lo.status_type_id = 999001
    and not exists (select 1 from mng_library_order_item loi
                            inner join mng_package_license pl on pl.package_license_id = loi.package_license_id
                            inner join mng_package p on p.package_id = pl.package_id
                            where loi.library_order_id = lo.library_order_id
                            and p.insert_user <> 'bakertaylor')
    and date_part('day', now() - lo.insert_date) >= 3
    "

show List

        		
        
    if argv.list
      data = []
      for job in dbreport.list()
        conf = dbreport.get job
        data.push
          job: job
          title: conf.title
          to: conf.email.to
      report = new Report()
      report.h1 "List of possible jobs:"
      report.table data, ['JOB', 'TITLE', 'TO']
      report.p 'Run them using their job name.'
      console.log()

      console.log()
